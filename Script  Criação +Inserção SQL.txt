    ID_CONTATO SERIAL PRIMARY KEY,
    CONTATO VARCHAR(50),
    FK_PESSOA_ID_PESSOA SERIAL,
    FK_TIPO_CONTATO_ID_TIPO_CONTATO SERIAL

    TIPO_CONTATO VARCHAR(50)

    MARCA VARCHAR(50)
);

CREATE TABLE CATEGORIA (
    ID_CATEGORIA SERIAL PRIMARY KEY,
    CATEGORIA VARCHAR(50)
);
